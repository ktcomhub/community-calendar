<script>

// 模擬活動數據 (你可以自行修改或擴展)

const events = [

{ date: '2026-03-05', title: '長者智能手機班', time: '10:00 - 12:00', desc: '教導街坊使用常用手機應用程式。', color: 'bg-blue-100 text-blue-800', link: 'https://forms.google.com/your-form-1' },

{ date: '2026-03-12', title: '手工皂工作坊', time: '14:30 - 16:30', desc: '一起製作天然手工皂，送禮自用兩相宜。', color: 'bg-purple-100 text-purple-800', link: 'https://forms.google.com/your-form-2' },

{ date: '2026-03-20', title: '社區共餐日', time: '18:00 - 20:00', desc: '每人帶一道菜，分享美味與故事。', color: 'bg-orange-100 text-orange-800', link: 'https://forms.google.com/your-form-3' },

{ date: '2026-04-05', title: '兒童畫畫比賽', time: '11:00 - 13:00', desc: '主題：我眼中的社區。', color: 'bg-green-100 text-green-800', link: 'https://forms.google.com/your-form-4' },

{ date: '2026-04-18', title: '義務理髮服務', time: '14:00 - 17:00', desc: '專業髮型師為街坊提供免費剪髮服務。', color: 'bg-red-100 text-red-800', link: 'https://forms.google.com/your-form-5' }

];

 

let currentViewDate = new Date(2026, 2, 1); // 預設顯示 2026年3月 (Month是從0開始)

 

function renderCalendar() {

const calendarDays = document.getElementById('calendarDays');

const monthYearLabel = document.getElementById('currentMonthYear');

 

calendarDays.innerHTML = '';

 

const year = currentViewDate.getFullYear();

const month = currentViewDate.getMonth();

 

monthYearLabel.innerText = `${year}年 ${month + 1}月`;

 

const firstDayOfMonth = new Date(year, month, 1).getDay();

const daysInMonth = new Date(year, month + 1, 0).getDate();

 

// 填充上個月的空白

for (let i = 0; i < firstDayOfMonth; i++) {

const emptyDiv = document.createElement('div');

emptyDiv.className = 'calendar-day border-r border-b border-slate-100 bg-slate-50/50';

calendarDays.appendChild(emptyDiv);

}

 

// 填充當月日期

for (let day = 1; day <= daysInMonth; day++) {

const dayDiv = document.createElement('div');

dayDiv.className = 'calendar-day border-r border-b border-slate-100 p-2 flex flex-col gap-1';

 

const dateStr = `${year}-${String(month + 1).padStart(2, '0')}-${String(day).padStart(2, '0')}`;

 

const dayNumber = document.createElement('span');

dayNumber.innerText = day;

dayNumber.className = 'text-sm font-medium text-slate-500';

dayDiv.appendChild(dayNumber);

 

// 找尋當日活動

const dayEvents = events.filter(e => e.date === dateStr);

dayEvents.forEach(event => {

const evTag = document.createElement('div');

evTag.className = `event-tag ${event.color} font-medium`;

evTag.innerText = event.title;

evTag.onclick = (e) => {

e.stopPropagation();

openModal(event);

};

dayDiv.appendChild(evTag);

});

 

calendarDays.appendChild(dayDiv);

}

}

 

function openModal(event) {

document.getElementById('modalTitle').innerText = event.title;

document.getElementById('modalDate').innerText = event.date;

document.getElementById('modalTime').innerText = event.time;

document.getElementById('modalDesc').innerText = event.desc;

document.getElementById('modalLink').href = event.link;

 

const modal = document.getElementById('eventModal');

modal.classList.remove('hidden');

modal.classList.add('flex');

}

 

function closeModal() {

const modal = document.getElementById('eventModal');

modal.classList.add('hidden');

modal.classList.remove('flex');

}

 

document.getElementById('prevBtn').onclick = () => {

currentViewDate.setMonth(currentViewDate.getMonth() - 1);

renderCalendar();

};

 

document.getElementById('nextBtn').onclick = () => {

currentViewDate.setMonth(currentViewDate.getMonth() + 1);

renderCalendar();

};

 

// 點擊 Modal 外部關閉

document.getElementById('eventModal').onclick = (e) => {

if (e.target.id === 'eventModal') closeModal();

};

 

// 初始化

window.onload = renderCalendar;

</script>
